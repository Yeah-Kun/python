function init_follow(){

            unfollowjs($(this).attr("data-id"),"?","switch");
          });

        else{
          $(this).text("[加关注]");
          $(this).bind("click",function(){
            followjs($(this).attr("data-id"));
          });
        }
      });
    }
  );
}
